1. Get the application URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Values.namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services operator-inventory)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Values.namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Values.namespace }} svc -w operator-inventory'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Values.namespace }} operator-inventory --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:8080
{{- else if contains "ClusterIP" .Values.service.type }}
  echo http://operator-inventory.{{ .Values.namespace }}.svc.cluster.local:8080
{{- end }}

2. Check the operator status:
  kubectl get pods --namespace {{ .Values.namespace }} -l app.kubernetes.io/name=inventory

3. View logs:
  kubectl logs --namespace {{ .Values.namespace }} -l app.kubernetes.io/name=inventory --tail=-1

{{- if .Values.metrics.enabled }}
4. Access metrics:
  kubectl port-forward --namespace {{ .Values.namespace }} svc/operator-inventory 8080:8080
  curl http://localhost:8080/metrics
{{- end }}
